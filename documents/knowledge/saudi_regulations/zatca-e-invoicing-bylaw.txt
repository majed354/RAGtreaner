لأولى: التعريفات

الســياق خلاف ذلــك:
الفاتورة الإلكترونية:1
فاتــورة ضريبيــة تصــدر بصيغــة إلكترونيــة منظمــة مــن خلال وســيلة إلكترونيــة. ولا تعــد الفاتــورة الورقيــة التــي
يتــم تحويلهــا إلــى صيغــة إلكترونيــة مــن خلال نســخها أو مســحها ضوئيــاً أو أي طريقــة أخــرى بأنهــا فاتــورة
إلكترونيــة لأغــراض هــذه اللائحــة.
2
الإشــعارات المدينــة والدائنــة التــي يجــب إصدارهــا وفقــاً لنظــام ضريبة القيمــة المضافــة ولائحته التنفيذيــة والتي

تصــدر بصيغــة إلكترونيــة منظمــة مــن خلال وســيلة إلكترونيــة. ولا تعــد الإشــعارات الورقيــة التــي يتــم تحويلهــا
إلــى صيغــة إلكترونيــة مــن خلال نســخها أو مســحها ضوئيــاً أو أي طريقــة أخــرى بأنهــا إشــعارات إلكترونيــة لأغــراض
هــذه اللائحــة.
3
ربــط أنظمــة إصــدار الفواتيــر الإلكترونيــة والإشــعارات الإلكترونيــة بأنظمــة الهيئــة بهــدف مشــاركة البيانــات
والمعلومــات وفقــاً لمــا يتــم تحديــده مــن ضوابــط ومتطلبــات فنيــة وقواعــد إجرائيــة يتــم اعتمادهــا واصدارهــا
بقــرار مــن المحافــظ اســتناداً لأحــكام هــذه اللائحــة.
4
الأشخاص الخاضعين لأحكام هذه اللائحة حسب ما هو منصوص عليه في المادة الثالثة منها
5

لائحــة الفوتــرة الإلكترونيــة الصــادرة بقــرار مجلــس إدارة الهيئــة بشــأن إصــدار الفواتيــر والإشــعارات الضريبيــة
إلكترونيــاً.
6
محافظ الهيئة العامة للزكاة والدخل
ب.فيمــا عــدا مــا نصــت عليــه الفقــرة (أ) مــن هــذه المــادة تكــون للألفــاظ والعبــارات الــواردة فــي هــذه اللائحــة
المعانــي المحــددة لهــا فــي الاتفاقيــة الموحــدة لضريبــة القيمــة المضافــة لــدول مجلــس التعــاون لــدول الخليــج
العربيــة ونظــام ضريبــة القيمــة المضافــة واللائحــة التنفيذيــة لنظــام ضريبــة القيمــة المضافــة.

الإحالة: لائحة الفوترة الإلكترونية، المادة الثانية
المادة الثانية:
لثانية: الغرض والنطاق
أ. تختــص هــذه اللائحــة بتحديــد الأحــكام والمتطلبــات والشــروط المتعلقــة بالفواتيــر الإلكترونيــة لأغــراض ضريبــة
القيمــة المضافــة بموجــب المــادة الثالثــة والخمســين مــن اللائحــة التنفيذيــة لضريبــة القيمــة المضافــة.
ب. تعد هذه اللائحة جزءًً لا يتجزأ من اللائحة التنفيذية لنظام ضريبة القيمة المضافة ومكملة لها.

الإحالة: لائحة الفوترة الإلكترونية، المادة الثالثة
المادة الثالثة:
لثالثة: الأشخاص الخاضعين للائحة
يخضع لهذه اللائحة كل مما يلي: أ.
الشخص الخاضع للضريبة المقيم في المملكة. 1
العميــل أو أي طــرف ثالــث يقــوم بإصــدار فاتــورة ضريبيــة نيابــة عــن الشــخص الخاضــع للضريبــة المقيــم فــي 2
المملكــة وفقــاً للائحــة التنفيذيــة لنظــام ضريبــة القيمــة المضافــة
ب. يجــب علــى الأشــخاص الخاضعيــن لهــذه اللائحــة المحدديــن فــي الفقــرة (أ) مــن هــذه المــادة، إصــدار فواتيــر
إلكترونيــة عــن جميــع تعاملاتهــم التــي يتوجــب إصــدار فواتيــر ضريبيــة عنها، بالإضافــة إلــى الإشــعارات الإلكترونية
التــي يجــب إصدارهــا فــي الحــالات المنصــوص عليهــا فــي نظــام ضريبــة القيمــة المضافــة ولائحتــه التنفيذيــة.

ج. لا يتوجــب علــى الأشــخاص غيــر المقيميــن فــي المملكــة إصــدار فواتيــر إلكترونيــة أو إشــعارات إلكترونيــة عــن
التوريــدات أو المبالــغ المســتلمة الخاضعــة للضريبــة فــي المملكــة.

الإحالة: لائحة الفوترة الإلكترونية، المادة الرابعة
المادة الرابعة:
لرابعة: الأحكام المتعلقة بالفواتير الإلكترونية والإشعارات الإلكترونية
أ. تعــد الفواتيــر الإلكترونيــة والإشــعارات الالكترونيــة الصــادرة بموجــب هــذه اللائحــة فواتيــر ضريبيــة وإشــعارات
دائنــة ومدينــة وفقــاً لمــا هــو محــدد فــي نظــام ضريبــة القيمــة المضافــة ولائحتــه التنفيذيــة، وينطبــق عليهــا
جميــع الأحــكام التــي تنطبــق علــى الفواتيــر الضريبيــة والإشــعارات الدائنــة والمدينــة ومنهــا، علــى ســبيل المثــال
لا الحصــر مــا يلــي:
1
القيمــة المضافــة
2
المنصــوص عليهــا فــي المــادة الثالثــة والخمســون مــن اللائحــة التنفيذيــة لنظــام ضريبــة القيمــة المضافــة
4
3
اللائحــة التنفيذيــة لنظــام ضريبــة القيمــة المضافــة
4

التنفيذيــة لنظــام ضريبــة القيمــة المضافــة
ب. باســتثناءً مــا ورد بشــأنه نــص خــاص بموجــب هــذه اللائحــة، تنطبــق الأحــكام المتعلقــة بإثبــات التعــاملات
الإلكترونيــة والتوقيعــات الإلكترونيــة المنصــوص عليهــا فــي نظــام التعــاملات الإلكترونيــة النافــذ فــي المملكــة
علــى الفواتيــر الإلكترونيــة والإشــعارات الإلكترونيــة الصــادرة بموجــب هــذه اللائحــة.
ج. للمحافــظ إصــدار أي أحــكام أو ضوابــط أو تفاصيــل إضافيــة يجــب توافرهــا في الفواتيــر الإلكترونية والإشــعارات
الإلكترونية .

الإحالة: لائحة الفوترة الإلكترونية، المادة لخامسة
المادة لخامسة:
لخامسة: المواصفات الفنية والقواعد الإجرائية
كحــد أدنــى- بالمعاييــر والمتطلبــات -أ. يجــب علــى الأشــخاص الخاضعيــن لأحــكام هــذه اللائحــة الالتــزام
والمواصفــات الفنيــة والقواعــد الإجرائيــة المنصــوص عليهــا فــي هــذه المــادة، فيمــا يتعلــق بالفواتيــر الإلكترونيــة
والإشــعارات الإلكترونيــة، وذلــك علــى النحــو التالــي:
1
الاتصــال بالإنترنــت
2
الربــط مــع أنظمــة خارجيــة باســتخدام واجهــة برمجــة التطبيقــات
3
للتلاعــب، وأن يتضمــن آليــة تتيــح كشــف أي تلاعــب قــد يتــم مــن قبــل المســتخدم أو أي طــرف آخــر
4
الاشــتراطات والضوابــط المطبقــة فــي المملكــة والمتعلقــة بأمــن البيانــات أوالمعلومــات أو الأمــن الســيبراني

ب. للمحافــظ تحديــد أيــة ضوابــط أو متطلبــات أو مواصفــات فنيــة أو قواعــد إجرائيــة إضافيــة تخــص الفواتيــر
الإلكترونيــة والإشــعارات الإلكترونيــة بمــا فيهــا تلــك المتعلقــة بإصدارهــا ومعالجتهــا وإرســالها وحفظهــا والربــط
بأنظمــة الفوتــرة الإلكترونيــة وتبــادل أو مشــاركة البيانــات أو المعلومــات المرتبطــة بهــا.
ج. لأغــراض هــذه المــادة، يشــمل الحــل التقنــي كل مــن الأجهــزة والبرامــج والشــبكات ووســائل الربــط وحفــظ
وتبــادل المعلومــات، المرتبطــة بالفواتيــر الإلكترونيــة والإشــعارات الإلكترونيــة.
5

الإحالة: لائحة الفوترة الإلكترونية، المادة السادسة
المادة السادسة:
لسادسة: أحكام عامة
أ. للمحافــظ صلاحيــة تحديــد المهــل الزمنيــة والفئــات المســتهدفة واصــدار القــرارات والتعليمــات اللازمــة لتنفيــذ
أحــكام هــذه اللائحــة.
ب. للمحافــظ إصــدار القــرارات المتعلقــة بمتطلبــات وضوابــط وتفاصيــل وإجــراءًات ربــط أنظمــة الفوتــرة الإلكترونيــة
خلال مــدة لا تزيــد عــن
المهــل الزمنيــة التــي تســبق الالــزام بتطبيــق الربــط.

ب. يمنــح الأشــخاص الخاضعيــن لهــذه اللائحــة مهلــة قدرهــا اثنــي عشــر
فــي الجريــدة الرســمية لتطبيــق الأحــكام الــواردة فــي الفقــرة (ب) مــن المــادة الثالثــة مــن هــذه اللائحــة والخاصــة
بإصــدار الفواتيــر الإلكترونيــة والإشــعارات الإلكترونية.
